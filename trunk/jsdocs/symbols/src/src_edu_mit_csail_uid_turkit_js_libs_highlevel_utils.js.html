  2 /**
  3  * Use this instead of <code>Math.random()</code>. This is not just a
  4  * shorthand. It is important that TurKit programs run the same way each time
  5  * they are executed. This function wraps <code>Math.random()</code> inside a
  6  * call to {@link TraceManager#once}.
  7  */
  8 function random() {
  9 	return once(function() {
 10 				return Math.random()
 11 			})
 12 }

 14 /**
 15  * Randomizes the order of the elements in <i>a</i>.
 16  */
 17 function shuffle(a) {
 18 	for (var i = 0; i < a.length; i++) {
 19 		var ii = Math.floor(random() * a.length)
 20 		var temp = a[i]
 21 		a[i] = a[ii]
 22 		a[ii] = temp
 23 	}
 24 	return a
 25 }

 27 /**
 28 	<p>Creates a webpage, puts it on S3, and returns the URL.
 29 	The webpage represents a HIT.
 30 	The webpage is created by starting with a <a href="/src/resources/task-template.html">template HTML page</a>.
 31 	The <code>html</code> supplied to this function in injected into the template.
 32 	If <code>bucketName</code> is not supplied,
 33 	then the result is placed on S3 in the bucket <code>your-aws-access-key-id.TurKit</code> with the key <code>md5-hash-of-content.html</code>.
 34 	The S3 webpage is made publicly accessible.
 35 	The URL to the S3 webpage is returned.
 36 	You can pass the URL to <code>s3.deleteObject</code> to remove it.</p>
 37 	<p>If you supply an array or comma-delimited-string for <code>blockWorkers</code>,
 38 	then the html page will use JavaScript to prevent those workers from completing the HIT.</p>
 39 	<p>If you give any elements the class "random", then those elements will be randomly permuted each time a user views the page.
 40 	This is good for randomizing the choices in voting tasks.</p>
 41 	*/
 42 function createWebpageFromTemplate(html, blockWorkers, bucketName) {
 43 	if (!blockWorkers) {
 44 		blockWorkers = ""
 45 	} else if (blockWorkers instanceof Array) {
 46 		blockWorkers = blockWorkers.join(",")
 47 	}
 48 	if (!bucketName) {
 49 		bucketName = javaTurKit.awsAccessKeyID + ".TurKit"
 50 	}	
 51 	var s = ("" + javaTurKit.taskTemplate).
 52 		replace(/___CONTENT___/, html).
 53 		replace(/___BLOCK_WORKERS___/, blockWorkers)
 54 	var key = Packages.edu.mit.csail.uid.turkit.util.U.md5(s) + ".html"
 55 	return s3.putString(bucketName, key, s)
 56 }

 58 /**
 59  * Changes the mode. Possible values include "offline", "sandbox" and "real".
 60  * <ul>
 61  * <li>"offline" will not let you access MTurk or S3.</li>
 62  * <li>"sandbox" will let you access the MTurk sandbox, and the real S3.</li>
 63  * <li>"real" mode accesses the real MTurk and the real S3.</li>
 64  * </ul>
 65  */
 66 function setMode(mode) {
 67 	javaTurKit.setMode(mode)
 68 	mturk.requesterService = javaTurKit.requesterService
 69 	s3.s3Service = javaTurKit.s3Service
 70 }
