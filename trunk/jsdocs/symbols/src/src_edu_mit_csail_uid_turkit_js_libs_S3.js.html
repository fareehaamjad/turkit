  2 /**
  3  * You probably want to use the global variable <code>s3</code>.
  4  * 
  5  * @class S3 contains wrappers around the JetS3t API for accessing Amazon's S3 storage service.
  6  * 
  7  * <p>
  8  * <code>s3</code> is a global instance of the S3 class.
  9  * </p>
 10  */
 11 function S3() {
 12 	this.s3Service = javaTurKit.s3Service
 13 }
 14 
 15 /**
 16  * A reference to an S3Service object in the JetS3t API.
 17  */
 18 S3.prototype.s3Service = null
 19 
 20 /**
 21 	Add public permissions to the given S3Object, and upload it to S3.
 22 	This function will create the bucket if it doesn't exist.
 23 	Returns the URL for the object.
 24 	
 25 	<p><code>bucketName</code> is optional.
 26 	If you do not provide it, the default bucket will be
 27 	<code>your-aws-access-key-id.TurKit</code></p>
 28  */
 29 S3.prototype.putObjectRaw = function(bucketName, s3Object) {
 30 	var bucket = this.s3Service.getOrCreateBucket(bucketName)
 31 	var acl = this.s3Service.getBucketAcl(bucket)
 32 	acl.grantPermission(org.jets3t.service.acl.GroupGrantee.ALL_USERS,
 33 		org.jets3t.service.acl.Permission.PERMISSION_READ)
 34 	s3Object.setAcl(acl)
 35 	this.s3Service.putObject(bucket, s3Object);
 36 	var url = this.getURL(bucket, s3Object)
 37 	database.query("ensure(null, ['__S3_Objects', " + json(url) + "], " + json({}) + ")")
 38 	if (verbose) {
 39 		print("S3 object put at: " + url)
 40 	}
 41 	return url
 42 }

 44 /**
 45  * Calls {@link S3#putObjectRaw} inside of {@link TraceManager#once}.
 46  */
 47 S3.prototype.putObject = function(bucketName, s3Object) {
 48 	return once(function() {
 49 				return s3.putObjectRaw(bucketName, s3Object)
 50 			})
 51 }

 53 /**
 54 	Creates an S3 URL given a <code>bucket</code> and a <code>key</code>.
 55  */
 56 S3.prototype.getURL = function(bucket, key) {
 57 	if ((typeof bucket) == "object") {
 58 		bucket = bucket.getName()
 59 	}
 60 	if ((typeof key) == "object") {
 61 		key = key.getKey()
 62 	}
 63 	return "http://s3.amazonaws.com/" + bucket + "/" + key
 64 }
 65 
 66 /**
 67 	Extracts the bucket name and key from an S3 URL. For instance, given <code>http://s3.amazonaws.com/hello/hi.txt</code>,
 68 	returns <code>{bucket:"hello", key:"hi.txt"}</code>.
 69  */
 70 S3.prototype.getBucketAndKey = function(url) {
 71     var a = url.match(/http:\/\/s3\.amazonaws\.com\/([^\/]+)\/(.*)/)
 72     if (!a) {
 73         a = url.match(/http:\/\/([^\/]+)\.s3\.amazonaws\.com\/(.*)/)
 74         if (!a) return
 75     }
 76     return {bucket : a[1], key : a[2]}
 77 }
 78 
 79 /**
 80 	Remove the object with the given <code>key</code> from the bucket with the given <code>bucketName</code>.
 81 	If only one parameter is supplied, it assumes it is an S3 URL, and attempts to extract the bucket name and key from that.
 82  */
 83 S3.prototype.deleteObjectRaw = function(bucketName, key) {
 84 	if (!key) {
 85 		var a = this.getBucketAndKey(bucketName)
 86 		bucketName = a.bucket
 87 		key = a.key
 88 	}
 89 	this.s3Service.deleteObject(bucketName, key)
 90 	
 91 	var url = this.getURL(bucketName, key)
 92 	verbosePrint("deleted S3 object at: " + url)
 93 	database.query("delete __S3_Objects[" + json(url) + "]")	
 94 }

105 /**
106 	Create a public object in S3 based on a string.
107 	Returns the URL for the object.
108  */
109 S3.prototype.putStringRaw = function(bucketName, key, stringData) {
110 	var o = new Packages.org.jets3t.service.model.S3Object(key, stringData)
111 	if (key.match(/\.html$/)) {
112 		o.setContentType("text/html")
113 	}
114 	return this.putObjectRaw(bucketName, o)
115 }

126 /**
127 	Create a public object in S3 based on a file.
128 	Returns the URL for the object.
129  */
130 S3.prototype.putFileRaw = function(bucketName, file) {
131 	if ((typeof file) == "string") {
132 		file = new java.io.File(file)
133 	}
134 	return this.putObjectRaw(bucketName, new Packages.org.jets3t.service.model.S3Object(file))
135 }

143 			})
