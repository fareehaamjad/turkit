  2 /**
  3  * You probably want to use the global variable <code>s3</code>.
  4  * 
  5  * @class S3 contains wrappers around the JetS3t API for accessing Amazon's S3 storage service.
  6  * 
  7  * <p>
  8  * <code>s3</code> is a global instance of the S3 class.
  9  * </p>
 10  */
 11 function S3() {
 12 	this.init()
 13 }
 14 
 15 /**
 16  * Initialize fields of the S3 class.
 17  */
 18 S3.prototype.init = function() {
 19 	this.s3Service = javaTurKit.s3Service
 20 	this.defaultBucketName = javaTurKit.awsAccessKeyID + ".TurKit"
 21 }
 22 
 23 /**
 24  * A reference to an S3Service object in the JetS3t API.
 25  */
 26 S3.prototype.s3Service = null
 27 
 28 /**
 29  * The default name used for buckets: <code>your-aws-access-key-id.TurKit</code>.
 30  */
 31 S3.prototype.defaultBucketName = null
 32 
 33 /**
 34 	Add public permissions to the given S3Object, and upload it to S3.
 35 	This function will create the bucket if it doesn't exist.
 36 	Returns the URL for the object.
 37 	
 38 	<p><code>bucketName</code> is optional.
 39 	If you do not provide it, the name {@link S3#defaultBucketName} will be used.</p>
 40 	
 41 	<p>If you provide only 1 parameter, it will be interpreted as the <code>s3Object</code>.</p>
 42  */
 43 S3.prototype.putObjectRaw = function(bucketName, s3Object) {
 44 	if (!s3Object && bucketName) {
 45 		this.putObjectRaw(null, bucketName)
 46 	}
 47 	if (!bucketName) {
 48 		bucketName = this.defaultBucketName
 49 	}
 50 
 51 	var bucket = this.s3Service.getOrCreateBucket(bucketName)
 52 	var acl = this.s3Service.getBucketAcl(bucket)
 53 	acl.grantPermission(org.jets3t.service.acl.GroupGrantee.ALL_USERS,
 54 		org.jets3t.service.acl.Permission.PERMISSION_READ)
 55 	s3Object.setAcl(acl)
 56 	this.s3Service.putObject(bucket, s3Object);
 57 	var url = this.getURL(bucket, s3Object)
 58 	database.query("ensure(null, ['__S3_Objects', " + json(url) + "], " + json({}) + ")")
 59 	if (verbose) {
 60 		print("S3 object put at: " + url)
 61 	}
 62 	return url
 63 }
 64 
 65 /**
 66  * Calls {@link S3#putObjectRaw} inside of {@link TraceManager#once}.
 67  */
 68 S3.prototype.putObject = function(bucketName, s3Object) {
 69 	return once(function() {
 70 				return s3.putObjectRaw(bucketName, s3Object)
 71 			})
 72 }

 74 /**
 75 	Creates an S3 URL given a <code>bucket</code> and a <code>key</code>.
 76  */
 77 S3.prototype.getURL = function(bucket, key) {
 78 	if ((typeof bucket) == "object") {
 79 		bucket = bucket.getName()
 80 	}
 81 	if ((typeof key) == "object") {
 82 		key = key.getKey()
 83 	}
 84 	return "http://s3.amazonaws.com/" + bucket + "/" + key
 85 }
 86 
 87 /**
 88 	Extracts the bucket name and key from an S3 URL. For instance, given <code>http://s3.amazonaws.com/hello/hi.txt</code>,
 89 	returns <code>{bucket:"hello", key:"hi.txt"}</code>.
 90  */
 91 S3.prototype.getBucketAndKey = function(url) {
 92     var a = url.match(/http:\/\/s3\.amazonaws\.com\/([^\/]+)\/(.*)/)
 93     if (!a) {
 94         a = url.match(/http:\/\/([^\/]+)\.s3\.amazonaws\.com\/(.*)/)
 95         if (!a) return
 96     }
 97     return {bucket : a[1], key : a[2]}
 98 }
 99 
100 /**
101 	Remove the object with the given <code>key</code> from the bucket with the given <code>bucketName</code>.
102 	If only one parameter is supplied, it assumes it is an S3 URL, and attempts to extract the bucket name and key from that.
103 	
104 	<p><code>bucketName</code> is optional.
105 	If you do not provide it, the name {@link S3#defaultBucketName} will be used.</p>
106  */
107 S3.prototype.deleteObjectRaw = function(bucketName, key) {
108 	if (!bucketName) {
109 		bucketName = this.defaultBucketName
110 	}
111 	if (!key) {
112 		var a = this.getBucketAndKey(bucketName)
113 		bucketName = a.bucket
114 		key = a.key
115 	}
116 	this.s3Service.deleteObject(bucketName, key)
117 	
118 	var url = this.getURL(bucketName, key)
119 	verbosePrint("deleted S3 object at: " + url)
120 	database.query("delete __S3_Objects[" + json(url) + "]")	
121 }

125  */
126 S3.prototype.deleteObject = function(bucketName, key) {
127 	return once(function() {
128 				return s3.deleteObjectRaw(bucketName, key)
129 			})

132 /**
133 	Create a public object in S3 based on a string.
134 	Returns the URL for the object.
135 	
136 	<p><code>bucketName</code> is optional.
137 	If you do not provide it, the name {@link S3#defaultBucketName} will be used.</p>
138 	
139 	<p><code>key</code> is optional.
140 	If you do not provide it, the key will be the md5 hash of the <code>stringData</code>,
141 	with an .html extension.</p>
142 	
143 	<p>If there is only 1 parameter, it will be interpreted as <code>stringData</code>.</p>
144  */
145 S3.prototype.putStringRaw = function(bucketName, key, stringData) {
146 	if (!stringData && !key && bucketName) {
147 		return this.putStringRaw(null, null, bucketName)
148 	}
149 	if (!bucketName) {
150 		bucketName = this.defaultBucketName
151 	}
152 	if (!key) {
153 		key = md5(stringData) + ".html"
154 	}
155 	
156 	var o = new Packages.org.jets3t.service.model.S3Object(key, stringData)
157 	if (key.match(/\.html$/)) {
158 		o.setContentType("text/html")
159 	} else {
160 		o.setContentType("text/plain")
161 	}
162 	return this.putObjectRaw(bucketName, o)
163 }

168 S3.prototype.putString = function(bucketName, key, stringData) {

170 				return s3.putStringRaw(bucketName, key, stringData)

174 /**
175 	Create a public object in S3 based on a file.
176 	Returns the URL for the object.
177 	
178 	<p><code>bucketName</code> is optional.
179 	If you do not provide it, the name {@link S3#defaultBucketName} will be used.</p>
180 	
181 	<p>If there is only 1 parameter, it will be interpreted as <code>file</code>.</p>
182  */
183 S3.prototype.putFileRaw = function(bucketName, file) {
184 	if (!file && bucketName) {
185 		return this.putFileRaw(null, bucketName)
186 	}
187 	if (!bucketName) {
188 		bucketName = this.defaultBucketName
189 	}
190 	
191 	if ((typeof file) == "string") {
192 		file = getFile(file)
193 	}
194 	return this.putObjectRaw(bucketName, new Packages.org.jets3t.service.model.S3Object(file))
195 }

197 /**
198  * Calls {@link S3#putFileRaw} inside of {@link TraceManager#once}.
199  */
200 S3.prototype.putFile = function(bucketName, file) {
201 	return once(function() {
202 				return s3.putFileRaw(bucketName, file)
203 			})
204 }
