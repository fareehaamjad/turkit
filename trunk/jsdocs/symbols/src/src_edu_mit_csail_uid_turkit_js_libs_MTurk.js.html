 21 /*
 22  * Throw an exception if we cannot spend the given amount of <i>money</i>, or
 23  * create the given number of <i>hits</i>, without violating our safety limits.
 24  * The <i>callbackBeforeCrash</i> will be called before an exception is thrown,
 25  * if an exception is about to be thrown.
 26  */
 27 MTurk.prototype.assertWeCanSpend = function(money, hits, callbackBeforeCrash) {
 28 	if (!javaTurKit.safety)
 29 		return
 30 
 31 	var safety = {
 32 		moneySpent : 0,
 33 		hitsCreated : 0
 34 	}
 35 	safety = database.query("return ensure('__safetyCounters', " + json(safety)
 36 			+ ")")
 37 	safety.moneySpent += money
 38 	safety.hitsCreated += hits
 39 	if (safety.moneySpent > javaTurKit.maxMoney) {
 40 		if (callbackBeforeCrash)
 41 			callbackBeforeCrash()
 42 		throw "TurKit has detected a safety violation: spending too much money."
 43 				+ "You need to increase your spending limit with TurKit (not with MTurk) to overcome this problem."
 44 	}
 45 	if (safety.hitsCreated > javaTurKit.maxHITs) {
 46 		if (callbackBeforeCrash)
 47 			callbackBeforeCrash()
 48 		throw "TurKit has detected a safety violation: creating too many HITs."
 49 				+ "You need to increase your hit limit with TurKit (not with MTurk) to overcome this problem."
 50 	}
 51 	safety = database.query("__safetyCounters = " + json(safety))
 52 }

 54 /**
 55  * Repeatedly tries to execute the given <i>func</i>, and returns the result
 56  * once it succeeds.
 57  */
 58 MTurk.prototype.keepTrying = function(func) {
 59 	sleep(0.05)
 60 	var waitTime = 0.1
 61 	for (var i = 0; i < 10; i++) {
 62 		try {
 63 			var ret = func()
 64 			return ret
 65 		} catch (e) {
 66 			if (/throttled/.exec("" + e)) {
 67 				verbosePrint("throttled")
 68 				sleep(waitTime)
 69 				waitTime += 0.1
 70 			} else {
 71 				throw e
 72 			}
 73 		}
 74 	}
 75 	// try one last time (this time don't catch the error)
 76 	return func()
 77 }

 88 /**
 89  * Creates a HIT. <i>params</i> is an object with the following properties:
 90  * <ul>
 91  * <li><b>title</b>: displayed in the list of HITs on MTurk.</li>
 92  * <li><b>description</b>: <b>desc</b> is also accepted. A slightly longer
 93  * description of the HIT, also shown in the list of HITs on MTurk</li>
 94  * <li><b>question</b>: a string of XML specifying what will be shown. <a
 95  * href="http://docs.amazonwebservices.com/AWSMechanicalTurkRequester/2008-08-02/index.html?ApiReference_QuestionFormDataStructureArticle.html">See
 96  * documentation here</a>. Instead of <i>question</i>, you may use the following special parameters:
 97  <ul>
 98  <li><b>url</b>: creates an external question pointing to this URL</li>
 99  <li><b>height</b>: (optional) height of the iFrame embedded in MTurk, in pixels (default is 600).</li>
100  </ul>
101  
102  </li>
103  * <li><b>reward</b>: how many dollars you want to pay per assignment for this
104  * HIT.
105  * </ul>
106  * The following properties are optional:
107  * <ul>
108  * <li><b>hitTypeId</b>: this may be given instead of the information above;
109  * usually it is not known or needed.</li>
110  * <li><b>keywords</b>: keywords to help people search for your HIT.</li>
111  * <li><b>assignmentDurationInSeconds</b>: default is 1 hour's worth of
112  * seconds.</li>
113  * <li><b>autoApprovalDelayInSeconds</b>: default is 1 month's worth of
114  * seconds.</li>
115  * <li><b>lifetimeInSeconds</b>: default is 1 week's worth of seconds.</li>
116  * <li><b>maxAssignments</b>: default is 1.</li>
117  * <li><b>requesterAnnotation</b>: default is no annotation.</li>
118  * <li><b>qualificationRequirements</b>: default is no requirements.</li>
119  * <li><b>minApproval</b>: minimum approval percentage. The appropriate
120  * requirement will be added if you supply a percentage here.</li>
121  * <li><b>responseGroup</b>: default is no response group.</li>
122  * </ul>
123  */
124 MTurk.prototype.createHITRaw = function(params) {
125 	if (!params)
126 		params = {}
127 
128 	if (params.HITTypeID)
129 		params.hitTypeId = params.HITTypeID
130 	if (params.hitTypeId) {
131 	} else {
132 		params.hitTypeId = null
133 		if (!params.title)
134 			throw "createHIT requires a title"
135 
136 		if (params.desc)
137 			params.description = params.desc
138 		if (!params.description)
139 			throw "createHIT requires a description"
140 
141 		if (!params.reward)
142 			throw "createHIT requires a reward"
143 
144 		if (!params.assignmentDurationInSeconds)
145 			params.assignmentDurationInSeconds = 60 * 60 // one hour
146 
147 		if (params.minApproval) {
148 			ensure(params, "qualificationRequirements", [])
149 					.push(new Packages.com.amazonaws.mturk.requester.QualificationRequirement(
150 							"000000000000000000L0",
151 							Packages.com.amazonaws.mturk.requester.Comparator.GreaterThanOrEqualTo,
152 							params.minApproval, null, false))
153 		}
154 	}
155 
156 	if (params.html) {
157 		if (!params.bucket) {
158 			params.bucket = javaTurKit.awsAccessKeyID.toLowerCase() + "-turkit"
159 		}
160 		var s = ("" + javaTurKit.taskTemplate).replace(/\[\[\[CONTENT\]\]\]/, params.html)
161 		var key = Packages.edu.mit.csail.uid.turkit.util.U.md5(s) + ".html"
162 		params.url = s3.putString(params.bucket, key, s)
163 		
164 		if (params.blockWorkers) {
165 			if ((typeof params.blockWorkers) != "string") {
166 				params.blockWorkers = params.blockWorkers.join(",")
167 			}
168 			params.url += "?blockWorkers=" + params.blockWorkers
169 		}
170 	}
171 	
172 	if (params.url) {
173 		if (!params.height)
174 			params.height = 600
175 
176 		params.question = '<ExternalQuestion xmlns="http://mechanicalturk.amazonaws.com/AWSMechanicalTurkDataSchemas/2006-07-14/ExternalQuestion.xsd">'
177 				+ '<ExternalURL>'
178 				+ escapeXml(params.url)
179 				+ '</ExternalURL>'
180 				+ '<FrameHeight>'
181 				+ params.height
182 				+ '</FrameHeight>'
183 				+ '</ExternalQuestion>'
184 	}
185 
186 	if (!params.question)
187 		throw "createHIT requires a question (or a url, or html)"
188 
189 	if (!params.lifetimeInSeconds)
190 		params.lifetimeInSeconds = 60 * 60 * 24 * 7 // one week
191 
192 	if (!params.maxAssignments)
193 		params.maxAssignments = 1
194 
195 	if (!params.autoApprovalDelayInSeconds)
196 		params.autoApprovalDelayInSeconds = null
197 
198 	if (javaTurKit.safety) {
199 		if (params.hitTypeId) {
200 			ensure(params, 'responseGroup', []).push('HITDetail')
201 		} else {
202 			this.assertWeCanSpend(params.reward * params.maxAssignments,
203 					params.maxAssignments)
204 		}
205 	}
206 
207 	if (!params.requesterAnnotation)
208 		params.requesterAnnotation = null
209 
210 	if (!params.responseGroup)
211 		params.responseGroup = null
212 
213 	if (!params.qualificationRequirements)
214 		params.qualificationRequirements = null
215 
216 	var hit = this.keepTrying(function() {
217 				return mturk.requesterService.createHIT(params.hitTypeId,
218 						params.title, params.description, params.keywords,
219 						params.question, params.reward,
220 						params.assignmentDurationInSeconds,
221 						params.autoApprovalDelayInSeconds,
222 						params.lifetimeInSeconds, params.maxAssignments,
223 						params.requesterAnnotation,
224 						params.qualificationRequirements, params.responseGroup);
225 			})
226 
227 	if (javaTurKit.safety) {
228 		if (params.hitTypeId) {
229 			this.assertWeCanSpend(parseFloat(hit.getReward().getAmount())
230 							* params.maxAssignments, params.maxAssignments,
231 					function() {
232 						mturk.disableHITRaw(hit)
233 					})
234 		} else {
235 		}
236 	}
237 
238 	var hitId = "" + hit.getHITId()
239 	verbosePrint("created HIT: " + hitId)
240 	var url = (javaTurKit.mode == "sandbox"
241 					? "https://workersandbox.mturk.com/mturk/preview?groupId="
242 					: "https://www.mturk.com/mturk/preview?groupId=")
243 			+ hit.getHITTypeId()
244 	verbosePrint("        url: " + url)
245 	database.query("ensure(null, ['__HITs', " + json(javaTurKit.mode + ":" + hitId) + "], " + json({url : url}) + ")")
246 	return hitId
247 }

336 				mturk.requesterService.extendHIT(hitId, moreAssignments,
337 						moreSeconds)
338 			})

351 /**
352 	Deletes the given <code>hit</code>.
353 	If there are any completed assignments that have not been approved or rejected, then they are approved.
354  */
355 MTurk.prototype.deleteHITRaw = function(hit) {
356 	var hitId = this.tryToGetHITId(hit)
357 	
358 	// first, try to disable the HIT
359 	try {
360 		this.keepTrying(function() {
361 			mturk.requesterService.disableHIT(hitId)
362 		})
363 		verbosePrint("disabled HIT: " + hitId)
364 	} catch (e) {
365 		try {	
366 			// check to see if we have already deleted it
367 			hit = this.getHIT(hitId)
368 			if (hit.hitStatus == "Disposed") {
369 				verbosePrint("already deleted HIT: " + hitId)
370 			} else {
371 				// ok, it must be "Reviewable"
372 				// (since we couldn't disable it, and it isn't already deleted)
373 				
374 				// first, approve all the assignments
375 				foreach(hit.assignments, function (a) {
376 					if (a.assignmentStatus == "Submitted")
377 						mturk.approveAssignmentRaw(a)
378 				})
379 				
380 				// next, dispose of the HIT
381 				this.keepTrying(function() {
382 					mturk.requesterService.disposeHIT(hitId)
383 				})
384 				verbosePrint("disposed HIT: " + hitId)
385 			}
386 		} catch (e) {
387 			if (/AWS\.MechanicalTurk\.HITDoesNotExist/.exec("" + e)) {
388 				verbosePrint("HIT not found: " + hitId)
389 			} else {
390 				throw e
391 			}
392 		}
393 	}
394 	
395 	database.query("delete __HITs[" + json(javaTurKit.mode + ":" + hitId) + "]")
396 }

521 /**
522  * Returns an object representing all of the information MTurk has on the given
523  * HIT, including the assigments, and the associated answer data. The returned
524  * object will have a value called <i>done</i> set to true iff all the pending
525  * assigments for this HIT have been completed.
526  * 
527  * <p>
528  * Note that the <i>answer</i> data structure associated with each assigment is
529  * simplified. It is recommended that you print out the result of this function
530  * using {@link json}, in order to know what it looks like for your specific
531  * situation.
532  * </p>
533  */
534 MTurk.prototype.getHIT = function(hit) {
535 	var hitId = this.tryToGetHITId(hit)
536 	var hit = this.keepTrying(function() {
537 				return mturk.requesterService.getHIT(hitId)
538 			})
539 	var hit = {
540 		hitId : "" + hit.getHITId(),
541 		hitTypeId : "" + hit.getHITTypeId(),
542 		title : "" + hit.getTitle(),
543 		description : "" + hit.getDescription(),
544 		keywords : "" + hit.getKeywords(),
545 		reward : "" + hit.getReward().getAmount(),
546 		question : hit.getQuestion(),
547 		maxAssignments : "" + hit.getMaxAssignments(),
548 		assignmentDurationInSeconds : hit.getAssignmentDurationInSeconds(),
549 		autoApprovalDelayInSeconds : hit.getAutoApprovalDelayInSeconds(),
550 		requesterAnnotation : hit.getRequesterAnnotation(),
551 		hitStatus : "" + hit.getHITStatus(),
552 		hitReviewStatus : "" + hit.getHITReviewStatus(),
553 		creationTime : hit.getCreationTime().getTimeInMillis(),
554 		expiration : hit.getExpiration().getTimeInMillis(),
555 		assignments : []
556 	}
557 	foreach(this.getAssignmentsForHIT(hitId), function(javaAssignment) {
558 				function getTime(t) {
559 					if (t != null)
560 						return t.getTimeInMillis()
561 				}
562 				var assignment = {
563 					assignmentId : "" + javaAssignment.getAssignmentId(),
564 					hitId : "" + javaAssignment.getHITId(),
565 					assignmentStatus : ""
566 							+ javaAssignment.getAssignmentStatus(),
567 					requesterFeedback : ""
568 							+ javaAssignment.getRequesterFeedback(),
569 					workerId : "" + javaAssignment.getWorkerId(),
570 					acceptTime : getTime(javaAssignment.getAcceptTime()),
571 					approvalTime : getTime(javaAssignment.getApprovalTime()),
572 					autoApprovalTime : getTime(javaAssignment
573 							.getAutoApprovalTime()),
574 					deadline : getTime(javaAssignment.getDeadline()),
575 					rejectionTime : getTime(javaAssignment.getRejectionTime()),
576 					submitTime : getTime(javaAssignment.getSubmitTime())
577 				}
578 
579 				// deal with answers
580 				// NOTE: we use the name "answer" to refer to the set of
581 				// answers, because that is how Mechanical Turk refers to them
582 				// in the
583 				// Assignment data structure
584 				assignment.answer = {}
585 				var answers = mturk.requesterService.parseAnswers(javaAssignment
586 						.getAnswer()).getAnswer()
587 				for (var ai = 0; ai < answers.size(); ai++) {
588 					var a = answers.get(ai)
589 
590 					var rhs = a.getFreeText()
591 					if (rhs) {
592 						rhs = "" + rhs
593 					}
594 					if (!rhs) {
595 						rhs = a.getUploadedFileKey()
596 						if (rhs) {
597 							rhs = {
598 								uploadedFileKey : "" + rhs,
599 								uploadedFileSizeInBytes : a
600 										.getUploadedFileSizeInBytes()
601 							}
602 						}
603 					}
604 					if (!rhs) {
605 						rhs = []
606 						var sels = a.getSelectionIdentifier()
607 						for (var si = 0; si < sels.size(); si++) {
608 							rhs.push("" + sels.get(si))
609 						}
610 						var o = a.getOtherSelectionText()
611 						if (o) {
612 							rhs.push("" + o)
613 						}
614 					}
615 
616 					assignment.answer["" + a.getQuestionIdentifier()] = rhs
617 				}
618 
619 				hit.assignments.push(assignment)
620 			})
621 	hit.done = (hit.hitStatus == "Reviewable")
622 			&& (hit.assignments.length == hit.maxAssignments)
623 	return hit
624 }

626 /**
627  * Returns information about the <code>hit</code> if it is done (see
628  * {@link MTurk#getHIT}), and throwing the "stop" exception if it is still in
629  * progress.
630  */
631 MTurk.prototype.waitForHIT = function(hit) {
632 	var hitId = this.tryToGetHITId(hit)
633 	return once(function() {
634 				var hit = mturk.getHIT(hitId);
635 				if (!hit.done) {
636 					stop()
637 				};
638 				verbosePrint("hit completed: " + hitId)
639 				return hit
640 			})
641 }

646 /**
647  * Collects multiple votes until a minimum number of necessary votes for a
648  * single choice is achieved. The <code>hit</code> supplied to this function
649  * must have its <i>maxAssignments</i> set to the minimum number of votes
650  * necessary for a single choice. This function will add assignments if
651  * necessary. This function relies on <code>extractVoteFromAnswer</code> to
652  * extract a String representing the choice given an answer data structure (see
653  * {@link MTurk#getHIT}).
654  * 
655  * <p>
656  * The return value is an object with the following entries:
657  * </p>
658  * <ul>
659  * <li><b>bestOption</b>: the String representing the choice with the most
660  * votes.</li>
661  * <li><b>totalVoteCount</b>: the number of votes received from turkers.</li>
662  * <li><b>voteCounts</b>: an object with key/value pairs representing
663  * choice/voteCounts.</li>
664  * <li><b>hit</b>: the HIT data structure representing the HIT on the final iteration of the voting process.</li>
665  * </ul>
666  */
667 MTurk.prototype.vote = function(hit, extractVoteFromAnswer) {
668 	var necessaryVoteCount = null
669 	var hitId = this.tryToGetHITId(hit)
670 	while (true) {
671 		var hit = this.waitForHIT(hitId)
672 
673 		if (necessaryVoteCount == null) {
674 			necessaryVoteCount = hit.maxAssignments
675 		}
676 
677 		var votes = {}
678 		foreach(hit.assignments, function(assignment) {
679 					var vote = extractVoteFromAnswer(assignment.answer)
680 					votes[vote] = ensure(votes, [vote], 0) + 1
681 				})
682 		var winnerVotes, winner
683 		[winnerVotes, winner] = getMax(votes)
684 
685 		if (winnerVotes >= necessaryVoteCount) {
686 			foreach(hit.assignments, function(assignment) {
687 						mturk.approveAssignment(assignment)
688 					})
689 			this.deleteHIT(hit)
690 			return {
691 				bestOption : winner,
692 				totalVoteCount : hit.assignments.length,
693 				voteCounts : votes,
694 				hit : hit
695 			}
696 		} else {
697 			this.extendHIT(hit, necessaryVoteCount - winnerVotes, null)
698 		}
699 	}
700 }

702 /**
703  * Works just like the JavaScript array sort function, except that this one can
704  * perform comparisons in parallel on MTurk by catching the "stop" exception
705  * which is thrown when waiting on an MTurk HIT using {@link MTurk#waitForHIT}.
706  */
707 MTurk.prototype.sort = function(a, comparator) {
708 	traceManager.pushFrame()
709 	try {
710 		var sortTree = traceManager.getFrameValue("sortTree")
711 		if (!sortTree)
712 			sortTree = {}
713 
714 		function insertIntoTree(index, tree) {
715 			if (tree.index == null) {
716 				tree.index = index
717 				return
718 			}
719 			var comp = comparator(a[index], a[tree.index])
720 			if (comp == 0) {
721 			} else if (comp < 0) {
722 				insertIntoTree(index, ensure(tree, ["left"], {}))
723 			} else {
724 				insertIntoTree(index, ensure(tree, ["right"], {}))
725 			}
726 		}
727 		var done = true
728 		foreach(a, function(e, i) {
729 					if (attempt(function() {
730 								insertIntoTree(i, sortTree)
731 							})) {
732 						traceManager.setFrameValue("sortTree", sortTree)
733 					} else {
734 						done = false
735 					}
736 				})
737 		if (!done)
738 			stop()
739 
740 		var newA = []
741 		function traverseTree(tree) {
742 			if (!tree)
743 				return
744 			traverseTree(tree.left)
745 			newA.push(a[tree.index])
746 			traverseTree(tree.right)
747 		}
748 		traverseTree(sortTree)
749 		return newA
750 	} finally {
751 		traceManager.popFrame()
752 	}
753 }
